# Stop the app (press Ctrl+C if running)

# Deactivate venv
deactivate

# Delete the venv folder
rmdir /s /q venv

# Create fresh venv
python -m venv venv

# Activate
venv\Scripts\activate

# Upgrade pip
python -m pip install --upgrade pip

# Install fresh
pip install -r requirements.txt

# Run
python app.py
